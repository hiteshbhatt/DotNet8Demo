public class VendoVenueDbContext : DbContext
{
    public DbSet<User> Users { get; set; }
    public DbSet<Role> Roles { get; set; }
    public DbSet<UserRole> UserRoles { get; set; }

    public DbSet<Vendor> Vendors { get; set; }
    public DbSet<VendorUser> VendorUsers { get; set; }
    public DbSet<VendorContact> VendorContacts { get; set; }
    public DbSet<Address> Addresses { get; set; }

    public DbSet<VendorOffering> VendorOfferings { get; set; }
    public DbSet<Venue> Venues { get; set; }
    public DbSet<Service> Services { get; set; }
    public DbSet<VenueAmenity> VenueAmenities { get; set; }
    public DbSet<VenueImage> VenueImages { get; set; }
    public DbSet<Pricing> Pricing { get; set; }
    public DbSet<TrustProfile> TrustProfiles { get; set; }

    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        base.OnModelCreating(modelBuilder);

        // Seed Roles
        modelBuilder.Entity<Role>().HasData(roles);

        // Seed Users
        foreach (var user in users)
            modelBuilder.Entity<User>().HasData(user);

        // Seed UserRoles
        foreach (var ur in userRoles)
            modelBuilder.Entity<UserRole>().HasData(ur);

        // Seed Vendors
        foreach (var v in vendors)
            modelBuilder.Entity<Vendor>().HasData(v);

        // Seed VendorContacts
        foreach (var c in new[] { vendor1Contact })
            modelBuilder.Entity<VendorContact>().HasData(c);

        // Seed Addresses
        foreach (var a in new[] { vendor1Address, venue1Address, service1Address })
            modelBuilder.Entity<Address>().HasData(a);

        // Seed Offerings
        foreach (var o in venues.Cast<VendorOffering>().Concat(services))
            modelBuilder.Entity<VendorOffering>().HasData(o);

        // Seed Amenities & Images
        foreach (var a in venues[0].Amenities)
            modelBuilder.Entity<VenueAmenity>().HasData(a);

        foreach (var i in venues[0].Images)
            modelBuilder.Entity<VenueImage>().HasData(i);

        // Seed Pricing
        modelBuilder.Entity<Pricing>().HasData(venue1Pricing, service1Pricing);

        // Seed TrustProfiles
        modelBuilder.Entity<TrustProfile>().HasData(trustProfiles[0]);
    }
}
